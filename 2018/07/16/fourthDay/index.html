萌新前端之旅第四天

第四天学习概述：

第四天学了块级作用域，let，var，const的区别，IIFE，浅拷贝，深复制。还学了数组的迭代和归并等。

let命令

http://es6.ruanyifeng.com/#docs/let

let基本用法

用let声明的变量，只会在其所在代码块内有效，例如

{
let a = 1;
var b = 1;
}
console.log(a); // a is no defined
console.log(b); // 1

{
a = 1; // ReferenceError
console.log(a); // ReferenceError

let a; // 在此之上全为暂时性死区
console.log(a); // undefined

a = 2;
console.log(a); // 2
}

{
let a = 1;
let a = 2 ; // ReferenceError
}

for( let i = 0 ; i < 6 ; i++){
…
}
console.log(i); // ReferenceError

// IIFE 写法
(function () {
var a= 1;
…
}());

// 块级作用域写法
{
let a = 1;
…
}

const a = Object.freeze({});

// 常规模式时，下面一行不起作用；
// 严格模式时，该行会报错
a.num = 1;

// 使用 reduce() 方法可以执行求数组里所有值之和的操作
var nums = [1,2,3,4];
var sum = nums.reduce(function(prev, cur, index, array) {
console.log(prev); // 依次输出 1， 3， 6
console.log(cur); // 依次输出 2， 3， 4
console.log(index); // 依次输出 1, 2, 3
console.log(array); // 每次都输出原数组 [1,2,3,4]
return prev + cur;
})
console.log(sum); // 10

var nums = [1,2,3,4];
var a= 5;
var sum = nums.reduce(function(prev, cur, index, array) {
console.log(prev); // 依次输出 5， 6， 8， 11
console.log(cur); // 依次输出 1， 2， 3， 4
console.log(index); // 依次输出 0, 1, 2, 3
console.log(array); // 每次都输出原数组 [1,2,3,4]
return prev + cur;
}, a)

console.log(sum); // 15
